function generateBingoNumbers() { let numbers = []; // ከ1 እስከ 100 ያሉ ቁጥሮችን በቅደም ተከተል መሙላት for (let i = 1; i <= 100; i++) { numbers.push(i); } // ቁጥሮቹን በዘፈቀደ ማቀላቀል (Shuffle) for (let i = numbers.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [numbers[i], numbers[j]] = [numbers[j], numbers[i]]; } return numbers; } // ቁጥሮቹን ለማመንጨት const bingoCardNumbers = generateBingoNumbers(); console.log(bingoCardNumbers);
